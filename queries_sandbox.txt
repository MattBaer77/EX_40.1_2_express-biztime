FROM invoices AS i
JOIN companies AS c
ON i.comp_code = c.code
WHERE id = 1;

SELECT c.code, c.name, c.description, i.id, i.comp_code, i.amt, i.paid, i.add_date, i.paid_date

LEFT JOIN invoices AS i
ON c.code = i.comp_code

SELECT c.code, c.name, i.code, i.industry_name
FROM companies AS c
JOIN company_industry AS x ON c.code = x.comp_code
JOIN industries AS i ON x.industry_code = i.code
WHERE c.code = 'ibm';
